\section{Logo Retrieval}

\begin{frame}
	\frametitle{Logo Retrieval}
	\framesubtitle{Lösungsalternativen}
     	\heading{Sliding window method}
	\begin{itemize}
		\item Langsam
		\item Nicht skalierungsinvariant
	\end{itemize}
	\bigskip	
     	\heading{SIFT oder HOG Feature Extraktion}
	\begin{itemize}
		\item Keine Detektion von unbekannten Logos nur Wiedererkennung
	\end{itemize}
	\bigskip
	\heading{CNN-basierter globaler Deskriptor}
	\begin{itemize}
		\item Von dem ganzen Bild z.B. Szene Wiedererkennung
		\item Ungeeignet für kleinere Objekte
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Logo Retrieval}
	\framesubtitle{Gewählte Lösung}
     	\heading{Entwicklung der Proposal-basierten CNNs}
	\begin{itemize}
		\item Regionen mit CNN
		\begin{itemize}
			\item Vorschläge für Objektpositionen von externen Systemen
			\item Berechnung der Deskriptoren von Regionen mit CNN individuell voneinander
		\end{itemize}
		\item Fast R-CNN
		\begin{itemize}
			\item Berechnung der Deskriptoren von Regionen gemeinsam mit Fully Convolutional Netzwerk
			\item Ausschneiden von Regionen aus dem Feature-Map mit RoIPooling Schicht
		\end{itemize}
		\item Faster R-CNN
		\begin{itemize}
			\item Vorschläge für Objektpositionen von netzinternen Mini-Netz (RPN)
			\item Das ganze System kann End-to-End trainiert werden
		\end{itemize}
	\end{itemize}
     	\heading{State-of-the-Art Lösungen verwenden Proposal-basierten CNN Systeme für Closed-Set Logo Retrieval}	
\end{frame}

\section{Das vorgestellte Logo Retrieval System}

\begin{frame}
	\frametitle{Logo Retrieval System}
     	\heading{Die Lösung ist in zwei Teilen aufgeteilt}
	\begin{itemize}
		\item Logo Detektion
		\begin{itemize}
			\item Der Detektor is trainiert alle Arten von Logo-Bilder zu erkennen
			\item Funktioniert ohne Anfragebilder
		\end{itemize}
		\item Logo Vergleich
		\begin{itemize}
			\item Die Feature Vektoren von Reigonen werden extrahiert
			\item Vergleich miteinander
		\end{itemize}
	\end{itemize}

	 \label{f:outline}
	\end{figure}
\end{frame}

\section{Ausblick}

\begin{frame}
	\frametitle{Ausblick}
	\heading{Textbasierte Logos}
	\begin{itemize}
		\item Überwiegende Mehrheit der Logos ist textbasiert
		\item Erweiterung des Systems mit Texterkennung Subsystem
	\end{itemize}
	\heading{Logo Tracking}
	\begin{itemize}
		\item Ergänzung bei fehlender Detektion 
		\item Reduzierung einmaliger Fehlklassifikation der getrackten Objekte
	\end{itemize}
	\heading{Mehr Daten}
	\begin{itemize}
		\item Die Größe der fusionierten Logo Datensätzen zu klein für Deep Learning
	\end{itemize}
	\heading{MAC - Maximum Activation of Convolutions}
	\begin{itemize}
		\item Gängige Feature Extraction Methode für Objekt- und Szene Retrieval
		\item Gesucht wird der maximum Wert von allen Channels, der Vektor wird als Feature benutzt
	\end{itemize}
\end{frame}
